def self.merge_thermal_storage(existing_tes, new_tes)
  existing_tes.its_size = add_values(existing_tes.its_size, new_tes.its_size)
  existing_tes.ptes_size = add_values(existing_tes.ptes_size, new_tes.ptes_size)

  return existing_tes
end
